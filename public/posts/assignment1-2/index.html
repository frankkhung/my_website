delay_sum <- as.data.frame(aggregate(`Min Delay` ~ Incident, ttc_delay, mean))


delay_sum$Incident <- factor(delay_sum$Incident, levels = delay_sum$Incident[order(delay_sum$`Min Delay`, decreasing = TRUE)])
